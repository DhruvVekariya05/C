SELECT PRO_NAME, PRO_PRICE
FROM product
WHERE PRO_PRICE >= 250
ORDER BY PRO_PRICE DESC, PRO_NAME ASC;



2).
Ans :-

SELECT PRO_NAME, PRO_PRICE
FROM product
ORDER BY PRO_PRICE ASC
LIMIT 1;


3).
Ans :-

SELECT PRO_COM, AVG(PRO_PRICE) AS average_price
FROM product
GROUP BY PRO_COM;



4).
Ans:-

SELECT AVG(PRO_PRICE) AS average_total_price
FROM product;
